    <!-- Firebase SDK -->

    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.0.1/firebase-app.js";

        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.0.1/firebase-auth.js";

        import { getFirestore, collection, addDoc, onSnapshot, query, deleteDoc, getDocs, doc, serverTimestamp } from "https://www.gstatic.com/firebasejs/11.0.1/firebase-firestore.js";


        // Environment Variables

        const firebaseConfig = JSON.parse(__firebase_config);

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'rockwash-dbc-demo';

        

        // Init Firebase

        const app = initializeApp(firebaseConfig);

        const auth = getAuth(app);

        const db = getFirestore(app);


        // UI References

        const feedContainer = document.getElementById('liveFeed');

        const addLogBtn = document.getElementById('addLogBtn');

        const clearLogBtn = document.getElementById('clearLogBtn');

        const syncStatus = document.getElementById('sync-status');


        const events = [

            { type: 'SCAN', msg: 'RFID tag detected at Intake Gate.' },

            { type: 'DBC', msg: 'GENESIS: Sample metadata locked.' },

            { type: 'WASH', msg: 'Automated wash protocol engaged (Stage 1).' },

            { type: 'DBC', msg: 'GATE: Process adherence verified.' },

            { type: 'IMG',  msg: 'Asset capture: 8k Macro-Photography saved.' },

            { type: 'DBC', msg: 'ANCHOR: Image SHA-256 injected to DBC.' },

            { type: 'XRF',  msg: 'Geochemical scan complete: Fe/Mg focus.' },

            { type: 'DBC', msg: 'FINAL COMMIT: Governance cycle complete.' }

        ];


        // 1. Mandatory Rule 3: Auth First Sequence

        const initAuth = async () => {

            try {

                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

                    await signInWithCustomToken(auth, __initial_auth_token);

                } else {

                    await signInAnonymously(auth);

                }

            } catch (err) {

                console.error("Authentication failed:", err);

                feedContainer.innerHTML = `<div class="text-red-500">Cloud Sync Error: Auth Failed. Please refresh.</div>`;

            }

        };


        // Start Auth

        initAuth();


        // 2. Mandatory Rule 3: Guard all operations inside onAuthStateChanged

        onAuthStateChanged(auth, (user) => {

            if (!user) {

                syncStatus.classList.add('hidden');

                return;

            }


            syncStatus.classList.remove('hidden');


            // Mandatory Rule 1: Strict collection path

            const logsRef = collection(db, 'artifacts', appId, 'public', 'data', 'governance_logs');

            

            // Listen for changes

            const unsubscribe = onSnapshot(logsRef, (snapshot) => {

                const logs = [];

                snapshot.forEach(doc => logs.push({ id: doc.id, ...doc.data() }));

                

                // Rule 2: Sort in Javascript memory

                logs.sort((a, b) => (a.timestamp?.seconds || 0) - (b.timestamp?.seconds || 0));


                renderLogs(logs);

            }, (error) => {

                // Mandatory error callback

                console.error("Firestore Permission/Snapshot Error:", error);

                if (error.code === 'permission-denied') {

                    feedContainer.innerHTML = `<div class="text-red-500">Sync Error: Access denied. Verifying credentials...</div>`;

                }

            });


            // Handle Emit

            addLogBtn.onclick = async () => {

                const event = events[Math.floor(Math.random() * events.length)];

                addLogBtn.disabled = true;

                addLogBtn.textContent = "Syncing...";

                

                try {

                    await addDoc(logsRef, {

                        ...event,

                        timestamp: serverTimestamp(),

                        user: user.uid

                    });
